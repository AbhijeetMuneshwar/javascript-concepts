JavaScript Interview Concepts with Code Examples

This file covers the most commonly asked JavaScript questions for interviews in 2026, with short explanations and practical examples.

1. What is Closure?

A closure is when a function remembers and can access variables from its outer scope even after the outer function has finished executing.

function outer() {
  let count = 0;
  return function inner() {
    count++;
    console.log(count);
  };
}

const counter = outer();
counter(); // 1
counter(); // 2

2. Explain Event Loop

The event loop handles asynchronous tasks by continuously checking the call stack and task queues.

console.log("Start");

setTimeout(() => console.log("Timeout"), 0);

Promise.resolve().then(() => console.log("Promise"));

console.log("End");
// Output: Start -> End -> Promise -> Timeout

3. var vs let vs const

var is function-scoped and hoisted; let and const are block-scoped.

if (true) {
  var a = 1;
  let b = 2;
  const c = 3;
}

console.log(a); // 1
// console.log(b); // ReferenceError
// console.log(c); // ReferenceError

4. Promise vs Async/Await

Promises represent asynchronous work; async/await makes asynchronous code look synchronous and easier to read.

function fetchData() {
  return new Promise((resolve) => {
    setTimeout(() => resolve("Data received"), 1000);
  });
}

fetchData().then((data) => console.log(data));

async function getData() {
  const data = await fetchData();
  console.log(data);
}

getData();

5. Call vs Apply vs Bind

These methods set the this value for a function. call and apply invoke immediately; bind returns a new function.

const person = {
  name: "Alice",
  greet(city, country) {
    console.log(`${this.name} lives in ${city}, ${country}`);
  }
};

person.greet.call({ name: "Bob" }, "Paris", "France");
person.greet.apply({ name: "Charlie" }, ["London", "UK"]);
const greetBob = person.greet.bind({ name: "Diana" });
greetBob("Tokyo", "Japan");

6. this Keyword

The value of this depends on how a function is called.

const obj = {
  name: "Rahul",
  show() {
    console.log(this.name);
  }
};

obj.show(); // Rahul

const fn = obj.show;
fn(); // undefined (in strict mode)

7. Event Delegation

Instead of attaching listeners to every child, attach one listener to a parent and handle events from children.

document.querySelector("#list").addEventListener("click", (e) => {
  if (e.target.tagName === "LI") {
    console.log(e.target.textContent);
  }
});

8. Debounce vs Throttle

Debounce delays execution until the event stops; throttle limits execution to a fixed interval.

function debounce(fn, delay) {
  let timer;
  return function (...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

function throttle(fn, limit) {
  let inThrottle;
  return function (...args) {
    if (!inThrottle) {
      inThrottle = true;
      setTimeout(() => {
        inThrottle = false;
        fn.apply(this, args);
      }, limit);
    }
  };
}

9. Shallow Copy vs Deep Copy

A shallow copy copies the top level; nested objects are still shared. A deep copy duplicates nested values too.

const person = { name: "Asha", address: { city: "Pune" } };

const shallowCopy = { ...person };
shallowCopy.address.city = "Mumbai";
console.log(person.address.city); // Mumbai

const deepCopy = JSON.parse(JSON.stringify(person));
deepCopy.address.city = "Delhi";
console.log(person.address.city); // Mumbai

10. map vs forEach

map returns a new array; forEach only iterates and returns undefined.

const nums = [1, 2, 3];

const doubled = nums.map(n => n * 2);
console.log(doubled); // [2, 4, 6]

nums.forEach(n => console.log(n));

11. null vs undefined

null means “no value” intentionally; undefined means a variable is declared but not assigned yet.

let a;
console.log(a); // undefined

let b = null;
console.log(b); // null

console.log(a === b); // false

12. == vs ===

== performs type coercion; === checks both value and type.

console.log(1 == "1");  // true
console.log(1 === "1"); // false

13. Hoisting

Hoisting moves declarations to the top of the scope during execution. Initializations stay in place.

console.log(x); // undefined
var x = 10;

console.log(y); // ReferenceError
let y = 20;

14. Scope Chain

JavaScript looks for variables in the current scope, then parent scopes, and so on until it reaches the global scope.

let globalVar = "Global";

function outer() {
  let outerVar = "Outer";
  function inner() {
    console.log(globalVar); // Global
    console.log(outerVar); // Outer
  }
  inner();
}

outer();

15. Prototype Inheritance

Objects inherit properties and methods from their prototype chain.

const animal = {
  eats: true
};

const dog = Object.create(animal);
dog.barks = true;

console.log(dog.eats); // true

16. Promise.all vs Promise.allSettled

Promise.all rejects as soon as one promise rejects. Promise.allSettled waits for all and returns their results.

Promise.all([
  Promise.resolve("A"),
  Promise.reject("B")
]).catch(err => console.log(err));

Promise.allSettled([
  Promise.resolve("A"),
  Promise.reject("B")
]).then(console.log);

17. Microtask vs Macrotask Queue

Microtasks (like promises) run before the next macrotask (like setTimeout).

setTimeout(() => console.log("macrotask"), 0);
Promise.resolve().then(() => console.log("microtask"));
// Output: microtask -> macrotask

18. Memory Leaks in JavaScript

Memory leaks happen when objects stay referenced longer than needed, preventing garbage collection.

let largeData = [];
function leak() {
  largeData.push(new Array(1000000).fill("x"));
}

// Avoid keeping unnecessary global references

19. Currying

Currying transforms a function that takes multiple arguments into a chain of functions that each take one argument.

function curry(a) {
  return function (b) {
    return function (c) {
      return a + b + c;
    };
  };
}

console.log(curry(1)(2)(3)); // 6

20. Polyfills

Polyfills add missing functionality to older browsers. Common examples include bind, map, reduce, and debounce.

// Polyfill for bind
if (!Function.prototype.bind) {
  Function.prototype.bind = function (context, ...args) {
    const fn = this;
    return function (...newArgs) {
      return fn.apply(context, [...args, ...newArgs]);
    };
  };
}

function greet(name) {
  console.log(`Hello ${name}`);
}

const bound = greet.bind(null, "Aman");
bound();

Bonus: Common Coding Questions

These are frequently asked in interviews because they test array, string, and object manipulation skills.

21. Reverse a String

Reversing a string is a common starter question. You can reverse the whole string or reverse each word separately.

let str = "I love my india";

function reverse(str) {
  return str.split("").reverse().join("");
}

console.log(reverse(str));

let result = str
  .split(" ")
  .map(word => word.split("").reverse().join(""))
  .join(" ");

console.log(result);

22. Find Duplicate Elements

Duplicate values can be detected by comparing the current index with the first index of the element.

const arr = [1, 2, 3, 2, 4, 1];
const duplicates = arr.filter((item, index) => arr.indexOf(item) !== index);

console.log(duplicates); // [2, 1]

23. Remove Duplicates

Set is the easiest way to remove duplicate values from an array.

const arr = [1, 2, 3, 2, 4, 1];
const unique = [...new Set(arr)];

console.log(unique); // [1, 2, 3, 4]

24. Find Maximum Number

This shows how to iterate through an array and track the highest value found.

const numbers = [5, 10, 15, 20, 8];
let max = numbers[0];

for (let i = 0; i < numbers.length; i++) {
  if (numbers[i] > max) {
    max = numbers[i];
  }
}

console.log(max); // 20

25. Count Character Occurrences

You can count a single character or count all characters in a string using loops or objects.

const str = "javascript";
const char = "a";
let count = 0;

for (let i = 0; i < str.length; i++) {
  if (str[i] === char) {
    count++;
  }
}

console.log(count); // 2

const result = {};
for (const ch of str) {
  result[ch] = (result[ch] || 0) + 1;
}

console.log(result);

26. Palindrome Check

A palindrome reads the same forward and backward. This solution compares a string with its reversed version.

function isPalindrome(str) {
  return str === str.split("").reverse().join("");
}

console.log(isPalindrome("madam")); // true
console.log(isPalindrome("hello")); // false

27. Flatten a Nested Array

Flattening converts a deeply nested array into a single-level array.

const arr = [1, [2, 3], [4, [5, 6]], 7];
const flattened = arr.flat(Infinity);

console.log(flattened); // [1, 2, 3, 4, 5, 6, 7]

function flatten(arr) {
  return arr.reduce((acc, curr) => {
    return acc.concat(Array.isArray(curr) ? flatten(curr) : curr);
  }, []);
}

console.log(flatten([1, [2, [3, [4]], 5]]));

28. Group Array by Property

Grouping arrays by a key is useful for dashboards, reports, and data transformation.

const users = [
  { name: "John", department: "IT" },
  { name: "Jane", department: "HR" },
  { name: "Mike", department: "IT" },
  { name: "Sara", department: "HR" }
];

const grouped = users.reduce((acc, user) => {
  const key = user.department;
  if (!acc[key]) {
    acc[key] = [];
  }
  acc[key].push(user);
  return acc;
}, {});

console.log(grouped);

29. Sum Total Income

reduce is often used to compute totals from an array of objects.

const users = [
  { name: "John", income: 5000 },
  { name: "Jane", income: 7000 },
  { name: "Mike", income: 6000 }
];

const totalIncome = users.reduce((total, user) => total + user.income, 0);
console.log(totalIncome); // 18000

30. Dynamic Property Access

This pattern lets you access nested object values using a string path.

const user = {
  name: "John",
  address: {
    city: "Pune",
    state: "Maharashtra"
  }
};

function getNestedValue(obj, path) {
  return path.split(".").reduce((acc, key) => acc?.[key], obj);
}

console.log(getNestedValue(user, "address.city")); // Pune
console.log(getNestedValue(user, "address.state")); // Maharashtra

31. Prime Number Check

A prime number is divisible only by 1 and itself. This example prints all prime numbers up to 50.

for (let num = 2; num < 50; num++) {
  let prime = true;

  for (let i = 2; i < num; i++) {
    if (num % i === 0) {
      prime = false;
      break;
    }
  }

  if (prime) {
    console.log(num);
  }
}
